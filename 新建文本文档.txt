OLD_EMAIL="pengpeng2044@gmail.com"
CORRECT_NAME="pyhui"
CORRECT_EMAIL="replycornucopia@163.com"

    export GIT_AUTHOR_EMAIL="$CORRECT_EMAIL"
fi
' --tag-name-filter cat -- --branches --tags
